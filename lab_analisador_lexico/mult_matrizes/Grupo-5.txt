def multiplicacao_de_matrizes(int n, int A, int B) {
	int i;
	int j;
	int k;
	int C = new int[n][n];
	for(i = 0; i < n; i = i + 1) {
		for(j = 0; j < n; j = j + 1) {
			C[i][j] = 0;
			for (k = 0; k < n; k = k + 1) {
				C[i][j] = C[i][j] + A[i][k] * B[k][j]
			}
		}
	}
	read C;
	return;
}
